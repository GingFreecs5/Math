	\subsection*{Exercice 1 (Équations du premier ordre)}
	Résoudre les équations différentielles suivantes sur $\mathbb{R}$ :
	\begin{enumerate}
		\item $y' - 2y = 0$
		\item $2y' + y = 3$
		\item Déterminer la solution $f$ de l'équation $y' + 3y = 6$ telle que $f(0) = 4$.
	\end{enumerate}
	
	\subsection*{Exercice 2 (Équations du second ordre)}
	Résoudre les équations différentielles suivantes sur $\mathbb{R}$ :
	\begin{enumerate}
		\item $y'' - 3y' + 2y = 0$
		\item $y'' + 4y' + 4y = 0$
		\item $y'' + 2y' + 5y = 0$
		\item Déterminer la solution $g$ de l'équation $y'' + y = 0$ telle que $g(0) = 1$ et $g'(0) = 2$.
	\end{enumerate}
	
	\subsection*{Exercice 3 (Problème de synthèse)}
	Soit l'équation différentielle :
	\[ (E) : y'' - 2y' + y = 0 \]
	\begin{enumerate}
		\item Résoudre l'équation $(E)$ sur $\mathbb{R}$.
		\item Déterminer la solution $f$ de $(E)$ vérifiant $f(0) = 1$ et $f'(0) = 0$.
		\item Soit $g$ la fonction définie sur $\mathbb{R}$ par $g(x) = f(x) - x$.
		Étudier les variations de $g$ et déterminer ses limites en $+\infty$ et $-\infty$.
	\end{enumerate}
